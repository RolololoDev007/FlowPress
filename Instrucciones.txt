Maquina Ubuntu
1- Actualizar Ubuntu
2- Instalar Docker y Docker Compose 
	apt update && apt upgrade -y
	apt install -y ca-certificates curl gnupg lsb-release
	install -m 0755 -d /etc/apt/keyrings 
	curl -fsSL https://download.docker.com/linux/ubuntu/gpg | \
	gpg --dearmor -o /etc/apt/keyrings/docker.gpg
	chmod a+r /etc/apt/keyrings/docker.gpg
	echo \
	"deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.gpg] \
	https://download.docker.com/linux/ubuntu \
	$(lsb_release -cs) stable" | \
	tee /etc/apt/sources.list.d/docker.list > /dev/null
	apt update
	apt install -y docker-ce docker-ce-cli containerd.io docker-buildx-plugin docker-compose-plugin
	docker --version
	docker compose version

	sudo adduser dockeruser
	sudo usermod -aG docker dockeruser
	sudo usermod -aG sudo dockeruser

3- Instalar nGinx
	apt update && apt upgrade -y
	apt install -y nginx
	systemctl status nginx
	
4- Instalar Cloudflared y configurarlo
	curl -fsSL https://pkg.cloudflare.com/cloudflare-main.gpg | sudo tee /usr/share/keyrings/cloudflare-main.gpg >/dev/null
	echo "deb [signed-by=/usr/share/keyrings/cloudflare-main.gpg] https://pkg.cloudflare.com/cloudflared noble main" | sudo tee /etc/apt/sources.list.d/cloudflared.list
	sudo apt update
	sudo apt install cloudflared -y

	cloudflared --version

	cloudflared tunnel login
	cloudflared tunnel create flowpress-tunnel
	cloudflared tunnel route dns flowpress-tunnel *.flowpress.site

	nano ~/.cloudflared/config.yml  (HAY QUE IDENTARLO ANTES DE GUARDAR)
		tunnel: flowpress-tunnel
		credentials-file: /root/.cloudflared/<UUID>.json

		ingress:
		  - hostname: "*.flowpress.site"
			service: http://localhost:80
		  - service: http_status:404


	4.1- Creamos la carpeta de las credenciales y movemos dichas credenciales a la carpeta en cuestión
	mkdir -p /etc/cloudflared
	mv ~/.cloudflared/<nombre-archivo-cred>.json /etc/cloudflared/flowpress.json
	cloudflared tunnel run "flowpress-tunnel"
	cloudflared service install
	
5- Instalar .NET Runtime
	apt update
	apt install -y wget apt-transport-https ca-certificates
	wget https://packages.microsoft.com/config/debian/12/packages-microsoft-prod.deb
	dpkg -i packages-microsoft-prod.deb
	apt update
	apt install -y dotnet-runtime-9.0

6- Instalar Supabase (BBDD)
	# Get the code
	git clone --depth 1 https://github.com/supabase/supabase

	# Make your new supabase project directory
	mkdir supabase-project

	# Tree should look like this
	# .
	# ├── supabase
	# └── supabase-project

	# Copy the compose files over to your project
	cp -rf supabase/docker/* supabase-project

	# Copy the fake env vars
	cp supabase/docker/.env.example supabase-project/.env

	# Switch to your project directory
	cd supabase-project
	
	# Generate and apply all secrets at once
	sh ./utils/generate-keys.sh
	
	# Pull the latest images
	docker compose pull

Maquina Ubuntu
8- Configurar nGinx
	rm /etc/nginx/sites-enabled/default
	chown -R dockeruser:www-data /etc/nginx/sites-available
	chmod -R 775 /etc/nginx/sites-available
	chown -R root:www-data /etc/nginx/sites-enabled
	chmod -R 775 /etc/nginx/sites-enabled
	usermod -aG www-data dockeruser
	nginx -t
	systemctl reload nginx

9- Dirigirnos a la carpeta en la que hemos copiado el Worker
10- Ejecutar el Worker 
	dotnet Worker.dll
