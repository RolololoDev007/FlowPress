Maquina Ubuntu
1- Actualizar Ubuntu
2- Instalar Docker y Docker Compose 
	apt update && apt upgrade -y
	apt install -y ca-certificates curl gnupg lsb-release
	install -m 0755 -d /etc/apt/keyrings 
	curl -fsSL https://download.docker.com/linux/ubuntu/gpg | \
	gpg --dearmor -o /etc/apt/keyrings/docker.gpg
	chmod a+r /etc/apt/keyrings/docker.gpg
	echo \
	"deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.gpg] \
	https://download.docker.com/linux/ubuntu \
	$(lsb_release -cs) stable" | \
	tee /etc/apt/sources.list.d/docker.list > /dev/null
	apt update
	apt install -y docker-ce docker-ce-cli containerd.io docker-buildx-plugin docker-compose-plugin
	docker --version
	docker compose version

	sudo adduser dockeruser
	sudo usermod -aG docker dockeruser
	sudo usermod -aG sudo dockeruser

3- Instalar nGinx
	apt update && apt upgrade -y
	apt install -y nginx
	systemctl status nginx
	
4- Instalar Cloudflared y configurarlo
	apt update && apt upgrade -y
	apt install -y curl gnupg lsb-release
	sudo mkdir -p --mode=0755 /usr/share/keyrings
	curl -fsSL https://pkg.cloudflare.com/cloudflare-main.gpg | \
	sudo tee /usr/share/keyrings/cloudflare-main.gpg >/dev/null
	echo "deb [signed-by=/usr/share/keyrings/cloudflare-main.gpg] \
	https://pkg.cloudflare.com/ $(lsb_release -cs) main" | \
	sudo tee /etc/apt/sources.list.d/cloudflare.list
	sudo apt update
	sudo apt install -y cloudflared
	cloudflared --version

	cloudflared tunnel login
	cloudflared tunnel create "nombretunnel"
	nano ~/.cloudflared/config.yml
		tunnel: nombretunnel
		credentials-file: /etc/cloudflared/flowpress.json

		ingress:
		  - hostname: "*.flowpress.site"
			service: http://localhost:80
		  - service: http_status:404

	cloudflared tunnel route dns "nombretunnel" midominio.com
	cloudflared tunnel run "nombretunnel"
	cloudflared service install
	
5- Instalar .NET Runtime
	apt update
	apt install -y wget apt-transport-https ca-certificates
	wget https://packages.microsoft.com/config/debian/12/packages-microsoft-prod.deb
	dpkg -i packages-microsoft-prod.deb
	apt update
	apt install -y dotnet-runtime-9.0
